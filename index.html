1
إنشاء جداول Supabase

افتح Supabase → SQL Editor → الصق هذا الكود لإنشاء جداول قائمة الانتظار والعملاء.

SQL — Supabase 
-- جدول قائمة الانتظار
CREATE TABLE waitlist (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  email       text NOT NULL UNIQUE,
  name        text,
  company     text,
  phone       text,
  source      text DEFAULT 'website',
  status      text DEFAULT 'pending',
  created_at  timestamptz DEFAULT now()
);

-- جدول العملاء (بعد التفعيل)
CREATE TABLE customers (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  email       text NOT NULL UNIQUE,
  name        text NOT NULL,
  company     text,
  plan        text DEFAULT 'starter',
  status      text DEFAULT 'active',
  created_at  timestamptz DEFAULT now()
);

-- تفعيل Row Level Security
ALTER TABLE waitlist ENABLE ROW LEVEL SECURITY;
ALTER TABLE customers ENABLE ROW LEVEL SECURITY;

-- سماح للجميع بالإضافة (للموقع)
CREATE POLICY "allow_insert" ON waitlist
  FOR INSERT WITH CHECK (true);

-- عرض فقط للمدمين
CREATE POLICY "admin_select" ON waitlist
  FOR SELECT USING (auth.role() = 'service_role');

FROM_EMAIL=hello@zainai.sa

لا تشارك ملف .env مع أحد ولا تضعه في GitHub. أضف .env.local إلى ملف .gitignore فوراً.

كود ربط الموقع بـ Supabase

أنشئ ملف lib/waitlist.js — هذا الكود يربط نموذج التسجيل بقاعدة البيانات.

JavaScript — lib/waitlist.js 
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY
)

// إضافة بريد إلكتروني لقائمة الانتظار
export async function addToWaitlist({ email, name, company }) {
  // التحقق من صحة البريد
  if (!email || !email.includes('@')) {
    return { success: false, error: 'البريد الإلكتروني غير صحيح' }
  }

  // الإضافة لقاعدة البيانات
  const { data, error } = await supabase
    .from('waitlist')
    .insert([{ email, name, company, source: 'website' }])
    .select()

  if (error) {
    // البريد مسجّل مسبقاً
    if (error.code === '23505') {
      return { success: false, error: 'هذا البريد مسجل مسبقاً!' }
    }
    return { success: false, error: 'حدث خطأ، حاول مجدداً' }
  }

  // إرسال بريد ترحيب
  await sendWelcomeEmail({ email, name })

  return { success: true, data }
}

// إرسال بريد ترحيب عبر Resend
async function sendWelcomeEmail({ email, name }) {
  try {
    await fetch('/api/send-welcome', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, name })
    })
  } catch (e) {
    console.error('فشل إرسال البريد:', e)
  }
}

import { Resend } from 'resend'

const resend = new Resend(process.env.RESEND_API_KEY)

export default async function handler(req, res) {
  const { email, name } = req.body

  const { error } = await resend.emails.send({
    from: `زين AI <hello@zainai.sa>`,
    to: email,
    subject: '🎉 أنت على القائمة! مرحباً بك في زين AI',
    html: welcomeTemplate(name)
  })

  if (error) return res.status(400).json({ error })
  res.status(200).json({ success: true })
}

ربط النموذج بالموقع

استبدل زر الموقع الحالي بهذا الكود لربطه مباشرة بقاعدة البيانات.

JavaScript — في صفحة الموقع 
import { addToWaitlist } from './lib/waitlist'

// استبدل دالة handleSignup الحالية بهذه
async function handleSignup() {
  const email = document.getElementById('email-input').value
  const btn   = document.getElementById('signup-btn')
  const msg   = document.getElementById('success-msg')
  const err   = document.getElementById('error-msg')

  // إظهار حالة التحميل
  btn.textContent = 'جارٍ التسجيل...'
  btn.disabled = true

  const result = await addToWaitlist({ email })

  if (result.success) {
    msg.style.display = 'block'
    document.getElementById('email-input').style.display = 'none'
    btn.style.display = 'none'
  } else {
    err.textContent = result.error
    err.style.display = 'block'
    btn.textContent = '← احجز مكانك الآن'
    btn.disabled = false
  }
}

بالتأكيد. جميع بياناتك تُخزَّن حصريًا على خوادم داخل المملكة العربية السعودية ومتوافقة مع نظام حماية البيانات الشخصية (PDPL) ومتطلبات ساما. لا تُشارك بياناتك مع أي طرف ثالث.
بعد اشتراكك، سيتواصل معك فريقنا خلال 24 ساعة لجدولة جلسة الإعداد. في المتوسط، تكون منصتك جاهزة للعمل خلال 7 إلى 14 يوماً من تاريخ البدء.
للباقات الشهرية لا يوجد عقد، يمكنك الإلغاء في أي وقت. الباقات السنوية تُوفّر 20% خصم وتلتزم لمدة سنة واحدة فقط.
ستتلقى تنبيهاً عند الوصول لـ 80% من الحد. يمكنك شراء محادثات إضافية بسعر مخفض أو الترقية للباقة الأعلى. لن تتوقف خدمتك فجأة أبداً.
نعم، نماذجنا مدرّبة على اللهجة السعودية والعربية الفصحى. الروبوتات تفهم السياق المحلي والمصطلحات التجارية السعودية وتُجيب بطريقة طبيعية تمامًا.
